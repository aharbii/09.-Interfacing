    while (1)
    {
        // VOLUME_CONTROLLER_Runnable();
        LCD_SetCursor(0, 0);
        u16 adc = ADC_Read(CH_0);
        LCD_WriteNumber(adc);
        LCD_SetCursor(1, 0);
        u16 pressure = MPX4115_ReadPressure();
        LCD_WriteNumber(pressure / 10);
        LCD_WriteChar('.');
        LCD_WriteNumber(pressure % 10);
        LCD_WriteString((u8 *)"     ");
    }
}
